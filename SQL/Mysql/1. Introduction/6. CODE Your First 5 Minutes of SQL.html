Step 1.

Step 2.

SELECT * FROM orders; 

Step 3.

SELECT * 
FROM products
ORDER BY Price DESC;

Step 4.

SELECT 
 customerName,
 COUNT(*) AS 'number of orders'
FROM customers
INNER JOIN orders
 ON orders.customerID = customers.customerID
GROUP BY customers.customerID;
